# 사용되는 json  데이터와 속성
[] 배열 안에 {} 객체로 아래의 속성/속성값으로 일정이 로딩된다.
1. title :    - 화면에 보일 일정 제목
2. start: '2023-01-06'    start: '2023-01-10T09:00:00',  시작일정 날짜/시간
3. end: '2023-01-10'       end: '2023-01-10T12:00:00'    마지막일정 날짜/시간
4. groupId: 999  - 반복된 일정 그룹화해서 처리할 때 사용..
5. url: 'http://google.com/' - 클릭시, 링크해서 로딩되는 페이지를 설정


매개변수 arg를 통해서 자동으로 설정되는 속성.

start: Tue Jan 03 2023 00:00:00 GMT+0900 (한국 표준시), ==> Date 형
end: Wed Jan 04 2023 00:00:00 GMT+0900 (한국 표준시), 
startStr: '2023-01-03', ==> 날짜만 문자열로 표현
endStr: '2023-01-04', 
allDay: true, …}  ==> allDay : true 전일,  allDay : false 시간일정

# 등록된 일정
allDay: true
backgroundColor: ""
borderColor: ""
end: null
endStr: ""
extendedProps : 추가 속성을 지정할 활용..
groupId: "" : 반복 일정시
id: "" : 핵심 key
start: Fri Jan 06 2023 00:00:00 GMT+0900 (한국 표준시)
startStr: "2023-01-06"
textColor: ""  : 글자 색상
title: "프로젝트 시작"

==> DB 처리..테이블..
create table calender(
	id number primary key,
	title varchar2(100),
	start1 varchar2(50), -- date의 ISO형식
	end1 varchar2(50),
	background_color varchar2(20),
	text_color varchar2(20),
	all_day number(1),  -- DB에서 0/1 ==> DTO false/true	
	url_link varchar2(200),
	writer varchar2(50),
	content varchar2(500)
);
create sequence cal_seq;
insert into calender values(cal_seq.nextval, '첫번째 일정', '2026-01-07','2026-01-08',
	'#009933','#ffff99',1,'http://www.naver.com','홍길동','일정 등록 연습' );
SELECT * FROM calender;
