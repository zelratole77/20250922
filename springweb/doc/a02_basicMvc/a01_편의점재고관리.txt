		- client  http://localhost:5050/basic01
		- controller  
				@GetMapping("basic01")
				public String basic01(){
					return "b01_basic\a01_basic01"
				}
				z01_basic/basic01.jsp 를 호출
				
				
				springweb\src\main\webapp\WEB-INF\views\b01_basic\a01_basic01.jsp
				==>
			# viewResolver 설정(jsp 경로 단축)
			# 접두사 Refix로 파일이 위치한 포더 경로
			spring.mvc.view.prefix=/WEB-INF/views/
			# 접미사 suffix : 파일 확장자명
			spring.mvc.view.suffix=.jsp
			# return "call"  ==> "/WEB-INF/views/call.jsp" 처리와 동일..
			# return "b01_basic\\a01_basic01"; ==> /WEB-INF/views/b01_basic/a01_basic01.jsp
